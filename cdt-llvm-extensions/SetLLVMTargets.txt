### set targets
# In LLVM 18, WebAssembly is a full target (not experimental).
# We build X86 + WebAssembly, plus the host architecture for native builds.
function(set_llvm_targets)
  # Determine host architecture target
  if(CMAKE_SYSTEM_PROCESSOR MATCHES "aarch64|arm64|ARM64")
    set(CDT_HOST_TARGET "AArch64")
  else()
    set(CDT_HOST_TARGET "X86")
  endif()

  # Build targets: host arch + WebAssembly (deduplicated if host is X86)
  # Include BPF and RISCV for future contract targets (issues #28, #29)
  if(CDT_HOST_TARGET STREQUAL "X86")
    set(CDT_LLVM_TARGETS "X86;WebAssembly;BPF;RISCV")
  else()
    set(CDT_LLVM_TARGETS "${CDT_HOST_TARGET};X86;WebAssembly;BPF;RISCV")
  endif()

  if(EXISTS ${LLVM_DEST}/llvm/CMakeLists.txt)
    file(READ ${LLVM_DEST}/llvm/CMakeLists.txt ROOT_LLVM_CMAKE_CONTENTS)
    string(FIND "${ROOT_LLVM_CMAKE_CONTENTS}"
           "set(LLVM_TARGETS_TO_BUILD \"all\""
           IDX_LLVM_BUILD_TARGETS)

    string(FIND "${ROOT_LLVM_CMAKE_CONTENTS}"
           "set(LLVM_TARGETS_TO_BUILD \"${CDT_LLVM_TARGETS}\""
           IDX_RESOLVED_LLVM_BUILD_TARGETS)

    if(IDX_LLVM_BUILD_TARGETS GREATER 0)
      string(REPLACE
           "set(LLVM_TARGETS_TO_BUILD \"all\""
           "set(LLVM_TARGETS_TO_BUILD \"${CDT_LLVM_TARGETS}\""
           NEW_ROOT_LLVM_CMAKE_CONTENTS
           "${ROOT_LLVM_CMAKE_CONTENTS}")
      file(WRITE ${LLVM_DEST}/llvm/CMakeLists.txt "${NEW_ROOT_LLVM_CMAKE_CONTENTS}")
    else()
      if(NOT IDX_RESOLVED_LLVM_BUILD_TARGETS GREATER 0)
        message(FATAL_ERROR,"Cannot find TARGETS_TO_BUILD in ${LLVM_DEST}/llvm/CMakeLists.txt")
      endif()
    endif()
  else()
    message(FATAL_ERROR,"Need to set targets, yet file does not exist ${LLVM_DEST}/llvm/CMakeLists.txt")
  endif()
endfunction()
