include(ExternalProject)
find_package(Git REQUIRED)
include(GNUInstallDirs)

# Build LLVM with native target support for the host architecture
if(CMAKE_SYSTEM_PROCESSOR MATCHES "aarch64|arm64|ARM64")
   set(CDT_LLVM_TARGETS "AArch64")
else()
   set(CDT_LLVM_TARGETS "X86")
endif()

# Pass host architecture to LLVM for Apple Silicon support
set(CDT_LLVM_EXTRA_ARGS "")
if(APPLE AND CMAKE_SYSTEM_PROCESSOR MATCHES "arm64|ARM64")
   set(CDT_LLVM_EXTRA_ARGS -DCMAKE_OSX_ARCHITECTURES=arm64)
endif()

ExternalProject_Add(
  CDTClang
  CMAKE_ARGS -DCMAKE_INSTALL_PREFIX=${CMAKE_INSTALL_PREFIX}/llvm -DCMAKE_CXX_COMPILER=${CMAKE_CXX_COMPILER} -DCMAKE_C_COMPILER=${CMAKE_C_COMPILER} -DCMAKE_BUILD_TYPE=Release -DCDT_TOOL_DIR=${CMAKE_SOURCE_DIR}/tools -DCMAKE_CXX_FLAGS=${CMAKE_CXX_FLAGS} -DLLVM_CCACHE_BUILD=Off -DLLVM_TARGETS_TO_BUILD=${CDT_LLVM_TARGETS} ${CDT_LLVM_EXTRA_ARGS}

  SOURCE_DIR "${CMAKE_SOURCE_DIR}/cdt-llvm"
  BINARY_DIR "${CMAKE_BINARY_DIR}/cdt-llvm"
  UPDATE_COMMAND ""
  PATCH_COMMAND  ""
  TEST_COMMAND   ""
  INSTALL_COMMAND ""
  BUILD_ALWAYS 1
)
